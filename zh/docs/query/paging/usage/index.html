                }
                storeName?.takeIf { it.isNotEmpty() }?.let {
                    where(table.store.name eq it)
                }
                orderBy(pageable.sort)
                select(
                    table.fetch(fetcher)
                )
            }

                pageable.pageNumber,
                pageable.pageSize
            )
}
